query getMakeSearch($queryInput: MakeSearchQueryInput!) {
  makeSearch(query: $queryInput) {
    id
    queryId
    queryText
    category {
      ...CategoryShortFragment
      __typename
    }
    categoryTree {
      category {
        ...CategoryFragment
        __typename
      }
      total
      __typename
    }
    items {
      catalogCard {
        __typename
        ...SkuGroupCardFragment
      }
      __typename
    }
    facets {
      ...FacetFragment
      __typename
    }
    total
    mayHaveAdultContent
    __typename
  }
}
